class MCPUnifiedRouter:
    """统一MCP工具暴露路由 - 修复G-001/002/003"""
    
    def __init__(self):
        self.tool_registry = {}  # 工具注册表
        self._register_all_mcp_tools()
    
    def _register_all_mcp_tools(self):
        """注册所有MCP工具到HTTP路由"""
        # 1. 注册memory-engine工具 (39个全部暴露)
        for tool in memory_engine_tools:  # 当前15个→目标39个
            self._register_tool("memory-engine", tool)
        
        # 2. 注册agent-framework工具 (当前0个→目标8个)
        for tool in agent_framework_tools:  # agent_dispatch/rule_evaluate等
            self._register_tool("agent-framework", tool)
        
        # 3. 注册command-executor/ops-engine/performance-profiler/security-scanner
        for server in ["command-executor", "ops-engine", "performance-profiler", "security-scanner"]:
            for tool in self._discover_tools(server):
                self._register_tool(server, tool)
    
    def _register_tool(self, server_name: str, tool: MCPTool):
        """单个工具注册"""
        route_path = f"/api/mcp/{server_name}/{tool.name}"
        self.tool_registry[route_path] = {
            "server": server_name,
            "tool": tool,
            "exposed": True,  # 关键：暴露标记
            "schema": tool.input_schema
        }
```

# core/search/multi_signal_retriever.py (新增)
class MultiSignalRetriever:
    """多信号检索器 - 融合Mem0核心创新"""
    
    async def retrieve(self, query: str, top_k: int = 5) -> List[Memory]:
        # 三路并行打分 (Mem0 2026.4算法)
        tasks = [
            self._semantic_score(query),    # 语义相似度
            self._keyword_score(query),     # 关键词匹配
            self._entity_score(query)       # 实体匹配
        ]
        semantic, keyword, entity = await asyncio.gather(*tasks)
        
        # 分数融合 (加权平均)
        fused_scores = self._fuse_scores(
            semantic, keyword, entity,
            weights=[0.5, 0.3, 0.2]  # Mem0推荐权重
        )
        
        # 返回top_k结果
        return self._get_top_k(fused_scores, top_k)
```
