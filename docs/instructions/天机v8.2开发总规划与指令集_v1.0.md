- [ ] **1.3.1**: 实现混合检索接口

  ```python
  class HybridRetriever:
      def __init__(self):
          self.vector_store = SQLiteFTS5()
          self.graph_store = TianjiGraphStore()
          self.embedder = DeepSeekEmbedder()
      
      def retrieve(self, query: str, top_k: int = 10) -> List[Dict]:
          """混合检索"""
          # Step 1: 向量检索 (入口点)
          vector_results = self.vector_store.search(query, limit=top_k)
          
          # Step 2: 图谱扩展 (深度)
          graph_context = []
          for result in vector_results:
              entities = self.extract_entities(result.content)
              for entity in entities:
                  # 多跳推理 (2跳)
                  related = self.graph_store.multi_hop(entity, hops=2)
                  graph_context.extend(related)
          
          # Step 3: 融合排序
          merged = self.merge_and_rank(vector_results, graph_context)
          
          return merged[:top_k]
  ```

  ```python
  class TripleRetriever:
      def __init__(self):
          self.vector = VectorRetriever()
          self.graph = GraphRetriever()
          self.keyword = KeywordRetriever()
      
      def retrieve(self, query: str, top_k: int = 10) -> List[Dict]:
          """三路召回"""
          # 并行检索
          with ThreadPoolExecutor(max_workers=3) as executor:
              future_vector = executor.submit(self.vector.retrieve, query, top_k)
              future_graph = executor.submit(self.graph.retrieve, query, top_k)
              future_keyword = executor.submit(self.keyword.retrieve, query, top_k)
          
          vector_results = future_vector.result()
          graph_results = future_graph.result()
          keyword_results = future_keyword.result()
          
          # 融合排序 (RRF)
          return self.reciprocal_rank_fusion(
              [vector_results, graph_results, keyword_results],
              top_k
          )
  ```
